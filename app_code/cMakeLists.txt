cmake_minimum_required(VERSION 3.28)

project(Car_identification_OCR)

# Set the C++ standard to C++17
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(sources cpp_file/main.cpp cpp_file/Ai.cpp cpp_file/Camera.cpp cpp_file/Display.cpp cpp_file/Image.cpp cpp_file/Notification.cpp cpp_file/Yolo_infer.cpp cpp_file/System.cpp)#indiquer les fichiers .cpp



add_executable(Car_identification_OCR ${sources})

find_package(OpenCV REQUIRED)
find_package(Torch REQUIRED)

target_include_directories(Car_identification_OCR PRIVATE hpp_file ${OpenCV_INCLUDE_DIRS} ${TORCH_INCLUDE_DIRS})

target_link_libraries(Car_identification_OCR ${OpenCV_LIBS} ${TORCH_LIBRARIES})
